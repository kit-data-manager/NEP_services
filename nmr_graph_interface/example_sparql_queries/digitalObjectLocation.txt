PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>
PREFIX fdoo: <https://datamanager.kit.edu/FDO-Graph#>

SELECT ?attributeValue_np ?pid
WHERE {
  fdoo:digitalObjectLocation fdoo:hasValue ?locationValue.
  FILTER regex(str(?locationValue), "b2share", "i")
  BIND(REPLACE(STR(?locationValue), "https://datamanager.kit.edu/FDO-Graph#", "") AS ?attributeValue_np)
  ?locationValue fdoo:isValueFor ?fdo.
  ?fdo rdfs:label ?pid.
}
